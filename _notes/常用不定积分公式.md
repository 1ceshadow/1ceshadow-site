$$
\begin{aligned}
& \qquad \text{微分} && \qquad \text{不定积分} \\
& \mathrm{d}(\ln x) = \dfrac{\mathrm{d}x}{x} && \int \dfrac{1}{x} \mathrm{d} x = \ln |x| +C \\
&\mathrm{d}(x^a) = a x^{a-1} \mathrm{d}x && \int x^a \mathrm{d}x = \dfrac{1}{a+1} x^{a+1}+C \quad (a \neq -1) \\
&\mathrm{d}(\tan x) = \sec ^2x \mathrm{d}x && \int \sec ^2x\mathrm{d}x=\tan x +C \\
&\mathrm{d}(\cot x) = -\csc^2x \mathrm{d}x && \int \csc^2x\mathrm{d}x = -\cot x +C \\
&\mathrm{d}(sec x) = \tan x \sec x \mathrm{d}x && \int \dots \\
&\mathrm{d}(\csc x) = -\cot x \csc x\mathrm{d}x && \int \dots \\
&\mathrm{d}(\arcsin x) = \dfrac{\mathrm{d}x}{\sqrt{1-x^2}} && \int \dots \\
&\mathrm{d}(\arctan x) = \dfrac{\mathrm{d}x}{1+x^2} && \int \dots \\
\end{aligned}
$$
其中$\int \dfrac{1}{x} \mathrm{d}x$ :
$$
\begin{align}
\int \dfrac{1}{x} \mathrm{d} x= \begin{cases}\text{When $x \geq 0$,} &\ln x+C \\
\text{When $x < 0$,} &\int -\dfrac{1}{-x}\mathrm{d}x = \int \dfrac{1}{-x} \mathrm{d}(-x)=\ln (-x) +C\\
\end{cases}
\end{align}
$$

---
补充一些：
$$
\begin{align}
&\int x^a \mathrm{d}x= \begin{cases} \dfrac{1}{a+1}x^{a+1} +C ,&\qquad a\neq-1 \\
\ln |x| +C ,&\qquad a=-1
\end{cases} \\
&\int \ln x \mathrm{d}x = x(\ln x -1) +C \\
\end{align}
$$



$\int \tan x \mathrm{d} x$ :
$$
\begin{align}
=\int \dfrac{\sin x}{\cos x} \mathrm{d} x = \int - \dfrac{1}{\cos x}\mathrm{d} \cos x=-\int\dfrac{1}{u} \mathrm{d}u = -\ln |\cos x|+C \\
\end{align}
$$
$\int \cot x \mathrm{d}x$ :
$$
\begin{align}
=\int\dfrac{\cos x}{\sin x} \mathrm{d}x=\ln |\sin x| +C\\
\end{align}
$$



$$
\begin{align}
&\int \sec x \mathrm{d}x = \ln |\sec x +\tan x| +C \qquad &&\int \csc x \mathrm{d}x = \ln |\csc x - \cot x|+C \\
&\int \sinh x \mathrm{d}x = \cosh x +C && \int \cosh x \mathrm{d}x = \sinh x +C \\
\end{align}
$$



$\int \dfrac{\mathrm{d}x}{\sqrt{a^2-x^2}}$ :
$$
\begin{align}
= \int \dfrac{\mathrm{d}\dfrac{x}{a}}{\sqrt{1-(\dfrac{x}{a})^2}} = \arcsin \dfrac{x}{a} +C \\
? = \int\dfrac{\mathrm{d} \sin u}{|\cos u|} = \int \dfrac{\cos u}{|\cos u|} \mathrm{d} u \;?
\end{align}
$$
$\int \dfrac{\mathrm{d}x}{\sqrt{x^2\pm a^2}}=\ln |x+\sqrt{x^2\pm a^2}|+C$ 稍微有些复杂（往细里写），它涉及一个进出根号的正负问题:
$$
\begin{align}
&=\int \dfrac{\mathrm{d}u}{\sqrt{u^2+1}} \;? \\
&=\begin{cases}\tan u = x/a(?), \quad \int \dfrac{\mathrm{d} \tan u}{\sqrt{\tan^2 u + 1}}=\int\dfrac{|\cos u|}{\cos^2u}\mathrm{d}u=\int \sec u \mathrm{d}u = \ln |\sec u + \tan u|+C = \\
\sec^2u=x/a(?), \quad\dots \\
\end{cases}
\end{align}
$$
$$
\sec x = \text{"}\pm \text{"} \sqrt{\dfrac{1}{\cos^2x}}=\text{"}\pm \text{"} \sqrt{\tan^2x+1}
$$
看了一些，关于这个提出来正负号的问题，不用考虑加正负号，因为这个是不定积分
比如$\sqrt{1+x^2}=x\sqrt{1+\frac{1}{x^2}}$ 
$$
\begin{align}
\int \sqrt{1+x^2} \mathrm{d}x &= \begin{cases}\int x\sqrt{1+\dfrac{1}{x^2}} \mathrm{d}x ,& \quad x>0 \quad (1)\\
\int -x \sqrt{1+x^2}\mathrm{d}x, & \quad x \leq 0 \quad (2)\\
\end{cases} \\
(1) &= \dfrac{1}{2}\int \sqrt{1+\dfrac{1}{x^2}} \mathrm{d} x^2
\end{align}
$$

$\int \sqrt{x^2 + a^2} \mathrm{d}x$ :
$$
\begin{align}
=\int a\sqrt{\frac{x^2}{a^2} + 1} \mathrm{d}x = a^2 \int \sqrt{(\tan \theta)^2+1}\mathrm{d}\tan \theta = a^2 \int \sec^3 \theta \mathrm{d} \theta \\ = a^2(\frac{1}{2}\sec \theta \tan \theta + \frac{1}{2}\ln |\sec \theta + \tan \theta| +C) \\(\tan \theta = \frac{x}{a}) \\
=\frac{x}{2} \sqrt{x^2+ a^2} +\frac{a^2}{2} \ln (x + \sqrt{x^2 + a^2})+C\\
\end{align}
$$
