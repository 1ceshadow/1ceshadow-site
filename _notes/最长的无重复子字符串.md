    char* end = s;

    while (*end != '\0') {
        end++;
    }

    int maxL = 0;
    for (char* i = s; i < end; i++) {
        int alphabet[128] = {0};
        int left = 1, right = 1, len = 1;
        alphabet[*i]++;

        // read left
        char* left_s = i - 1;
        while (left_s >= s && left == 1) {
            if (alphabet[*left_s - 0] != 0) {
                alphabet[*left_s - 0]++;
                len++;
                left_s--;
            }
            else {
                left = 0;
            }
        }
        // read right
        char* right_s = i + 1;
        while (right_s < end && right == 1) {
            if (alphabet[*right_s - 0] != 0) {
                alphabet[*right_s - 0]++;
                len++;
                right_s++;
            }
            else {
                right = 0;
            }
        }

        // update maxL
        if (maxL < len) {
            maxL = len;
        }
    }
    
    return maxL;
}
